import os
from typing import Optional

class SecretsManager:
    """Centralized pattern for loading API keys from environment only."""
    
    @staticmethod
    def get_api_key(key_name: str, default: Optional[str] = None) -> str:
        """Load API key from environment variable. Never defaults to hardcoded values."""
        value = os.getenv(key_name, default)
        if not value:
            raise ValueError(f"Required API key '{key_name}' not found in environment")
        return value

# API Configuration - All from environment
API_KEY_STRIPE = SecretsManager.get_api_key('API_KEY_STRIPE')
